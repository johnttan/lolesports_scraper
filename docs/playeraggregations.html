# 
def calctotalscore(player, score):
    playername = player['playername']
    return cGames.aggregate([
                {
                '$match': {'playerlist': player['playername']}
                },
                {
                '$project': {
                            score + 'score': '$players.' + playername + '.score.' + score
                            }
                },
                {
                '$group': {
                            '_id':playername,
                            'total' + score: {'$sum':'$' + score + 'score'}
                            }
                }
        ])
#
